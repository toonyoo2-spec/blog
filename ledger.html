/* 🌟 캘린더 렌더링 (DB의 color 및 owner 값 동적 바인딩 완벽 보장) */ function renderCalendar(entries){ const y = currentYear; const m = currentMonth - 1; const byDate = {}; entries.forEach(it => { if(!it.date) return; let dayKey = String(it.date).trim(); if(dayKey.includes('T')) dayKey = dayKey.split('T')[0]; const parts = dayKey.split('-'); if(parts.length === 3){ const dayNum = parseInt(parts[2], 10); (byDate[dayNum] = byDate[dayNum] || []).push(it); } }); const first = new Date(y, m, 1); const startOffset = first.getDay(); const daysInMonth = new Date(y, m + 1, 0).getDate(); const daysInPrevMonth = new Date(y, m, 0).getDate(); const today = new Date(); const isCurrentMonthToday = (today.getFullYear() === y && today.getMonth() === m); const todayNum = today.getDate(); const cells = []; const dowLabels = ['일','월','화','수','목','금','토']; dowLabels.forEach(l => cells.push(`
${l}
`)); for(let i = startOffset - 1; i >= 0; i--){ cells.push(`
${daysInPrevMonth - i}
`); } for(let d = 1; d <= daysInMonth; d++){ const isToday = isCurrentMonthToday && (d === todayNum); const dayEntries = byDate[d] || []; const itemsHtml = dayEntries.map(it => { const typeStr = String(it.type || "").trim(); const ownerStr = String(it.owner || "").trim(); const detailStr = escapeHtml(it.detail || it.small || it.large || "기타"); const amt = Number(it.amount) || 0; // 🎨 1. DB의 color 값을 최우선 사용 let bgColor = it.color ? String(it.color).trim() : null; // 🎨 2. DB에 color 값이 없을 경우 owner 및 type 조건부 자동 매핑 if (!bgColor) { if (typeStr === "수입") { bgColor = "#1976D2"; // 파란색 (수입) } else { switch (ownerStr) { case "관": bgColor = "#2E7D32"; break; // 🟢 초록 case "보라": bgColor = "#FFCDD2"; break; // 🌸 연분홍 case "자동차": bgColor = "#37474F"; break; // ⬛ 다크 차콜 case "공동": default: bgColor = "#512DA8"; break; // 🟣 보라색 } } } // 🎨 3. 연분홍(#FFCDD2) 배경일 때는 어두운 글자(#D32F2F), 그 외에는 하얀색(#FFFFFF) const isLightBg = bgColor.toUpperCase() === "#FFCDD2" || ownerStr === "보라"; const textColor = isLightBg && typeStr !== "수입" ? "#D32F2F" : "#FFFFFF"; return `
${detailStr} ${formatMoney(amt)}원
`; }).join(''); cells.push(`
${d}
${itemsHtml}
`); } const totalSoFar = startOffset + daysInMonth; const trailing = (7 - (totalSoFar % 7)) % 7; for(let d = 1; d <= trailing; d++){ cells.push(`
${d}
`); } document.getElementById('ledgerCalGrid').innerHTML = cells.join(''); }
